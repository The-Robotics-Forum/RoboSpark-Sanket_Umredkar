#include<iostream>
#include<string.h>
using namespace std;
 class Student
 {
     public:
     int roll;
     string name;
     float cgpa;
     int year;
     float *marks = new float[5];

    Student(){
        roll = 1;
        name = "Ashwin";
        year = 2;
    }
    int calculateCGPA()
    {
        float sum=0;
        for(int i=0;i<5;i++)
        sum+=marks[i];
        cgpa = (sum/500)*10;
        delete[] marks;
    }

    void getData()
    {
        cout<<"Enter Details of Student\nEnter Roll No : ";
        cin>>roll;
        cout<<"Enter Name : ";
        cin>>name;
        cout<<"Enter Year : ";
        cin>>year;
        for(int i=0;i<5;i++)
            {
            cout<<"Enter Mark of Subject "<<i+1<<" : ";
            cin>>marks[i];
            }
    }
    void showData()
    {
        calculateCGPA();
        cout<<"\nDetails are :\nRoll No : "<<roll<<"\nName : "<<name<<"\nYear : "<<year<<"\nCGPA : "<<cgpa;
    }
 };

 int main()
 {
     Student s;
     s.getData();
     s.showData();
 }
