#include<stdio.h>
#include<stdlib.h>
struct student
{
    char name[50];
    int id;
    float cgpa;
} s[10];

int main()
   {
    int i;
    printf("Enter information of students:\n");

    for (i = 0; i < 5; ++i)
    {
        s[i].id = i + 1;
        printf("\nEnter Id: ");
        scanf("%d", &s[i].id);
        printf("Enter Name: ");
        scanf("%s", s[i].name);
        printf("Enter CGPA: ");
        scanf("%f", &s[i].cgpa);
    }


    printf("\n\nDisplaying Information:\n\n");


    for (i = 0; i < 5; ++i)
    {
        printf("\nId number: %d\n", s[i].id);
        printf("Name: ");
        puts(s[i].name);
        printf("CGPA: %.1f", s[i].cgpa);
        printf("\n");
    }
    return 0;
}
